package io.github.kotindia

/**
 * Structured, type-safe context attached to every [ValidationResult.Invalid].
 *
 * Carries diagnostic parameters — never strings — so consumers can render
 * their own copy in any language with full parameter access.
 *
 * @sample io.github.kotindia.samples.validationContextSample
 */
public sealed interface ValidationContext {
    /**
     * No additional context. Default for [ValidationResult.Invalid] when no
     * structured parameters are meaningful (e.g., [InvalidReason.EMPTY]).
     */
    public data object None : ValidationContext

    /**
     * The input length did not match the expected length.
     *
     * @param expected The validator's required length.
     * @param actual The length of the input after normalisation.
     */
    public data class LengthMismatch(
        public val expected: Int,
        public val actual: Int,
    ) : ValidationContext
}

```kotlin
// Copyright (c) The KotIndia Authors. Licensed under the Apache License, Version 2.0.
// SPDX-License-Identifier: Apache-2.0

package io.github.kotindia

/**
 * The result of [validateProgressive] — a state-machine view of input that
 * may still be in progress.
 *
 * Designed for incremental UI input where validation runs on every keystroke.
 * Consumers render different UI per state (placeholder / typing / valid / error)
 * without writing the "is the field complete enough to validate?" glue code.
 *
 * **State semantics:**
 * - [Empty]  — input is blank or whitespace only.
 * - [Typing] — input has the right character class but is shorter than `maxLength`.
 *              Carries [Typing.visualText] which is the input formatted as far as
 *              the validator can — e.g., a 7-digit Aadhaar in progress is `"1234 567"`.
 * - [Valid]  — input is complete and passes ALL validation (format + checksum).
 * - [Invalid]  — input is **structurally** broken (wrong char class, beyond max length).
 *               Mid-typing inputs of the right char class do NOT return [Invalid].
 *
 * @sample io.github.kotindia.samples.progressiveResultSample
 */
public sealed interface ProgressiveResult {
    /** Input is empty or whitespace only. */
    public data object Empty : ProgressiveResult

    /**
     * Input is in progress — has the right character class, length less than `maxLength`.
     *
     * @param visualText The input formatted as far as the validator can render it
     *   for user display. May equal the raw input if no formatting is meaningful.
     */
    public data class Typing(
        public val visualText: String,
    ) : ProgressiveResult

    /** Input is complete and passes all validation. */
    public data object Valid : ProgressiveResult

    /**
     * Input is structurally invalid — wrong character class or beyond `maxLength`.
     *
     * @param reason See [InvalidReason].
     * @param context Typed parameters. See [ValidationContext].
     */
    public data class Invalid(
        public val reason: InvalidReason,
        public val context: ValidationContext = ValidationContext.None,
    ) : ProgressiveResult
}
```

- [ ] **14.10: Implement progressive API in `Aadhaar.kt`**

Add to `core/src/commonMain/kotlin/io/github/kotindia/Aadhaar.kt` (inside the `Aadhaar` object, after the existing `mask` method):

```kotlin
    /**
     * The exact length a valid Aadhaar number must have. Aadhaar is always 12 digits.
     */
    public val maxLength: Int get() = EXPECTED_LENGTH

    /**
     * The set of characters allowed in an Aadhaar input. Aadhaar is digits-only.
     *
     * Note: this is the **input filter**, not the format check. Spaces are stripped
     * during normalisation but are not in `allowedChars` because consumers should
     * filter input to `allowedChars` for keyboard restriction (e.g. Compose
     * `KeyboardOptions.NumberPassword`), then let [validate] handle whitespace.
     */
    public val allowedChars: CharRange get() = '0'..'9'

    /**
     * Strips any character not in [allowedChars] from [rawInput] and caps the
     * result at [maxLength] characters.
     *
     * Useful for `onValueChange` callbacks in incremental input UI:
     *
     * ```kotlin
     * onValueChange = { input -> text = Aadhaar.sanitize(input) }
     * ```
     *
     * @param rawInput Any user-typed string.
     * @return A digits-only string at most [maxLength] characters long.
     * @sample io.github.kotindia.samples.aadhaarSanitizeSample
     */
    public fun sanitize(rawInput: String): String =
        rawInput.filter { it in allowedChars }.take(maxLength)

    /**
     * State-machine validator for incremental input. See [ProgressiveResult] for state semantics.
     *
     * Treats input differently from [validate]:
     * - Whitespace-only input → [ProgressiveResult.Empty].
     * - Partial input (right char class, length < [maxLength]) → [ProgressiveResult.Typing].
     * - Complete input that passes all checks → [ProgressiveResult.Valid].
     * - Wrong character class or beyond [maxLength] → [ProgressiveResult.Invalid].
     *
     * @param value Raw or partial Aadhaar input.
     * @return [ProgressiveResult] state.
     * @sample io.github.kotindia.samples.aadhaarValidateProgressiveSample
     */
    public fun validateProgressive(value: String): ProgressiveResult {
        val normalized = value.trim().replace(Regex("\\s"), "")
        if (normalized.isEmpty()) return ProgressiveResult.Empty
        if (normalized.any { it !in allowedChars }) {
            return ProgressiveResult.Invalid(reason = InvalidReason.INVALID_FORMAT)
        }
        if (normalized.length > maxLength) {
            return ProgressiveResult.Invalid(
                reason = InvalidReason.WRONG_LENGTH,
                context = ValidationContext.LengthMismatch(expected = maxLength, actual = normalized.length),
            )
        }
        if (normalized.length < maxLength) {
            return ProgressiveResult.Typing(visualText = formatPartial(normalized))
        }
        // length == maxLength → fall through to full validation
        return when (val r = validate(normalized)) {
            is ValidationResult.Valid -> ProgressiveResult.Valid
            is ValidationResult.Invalid -> ProgressiveResult.Invalid(reason = r.reason, context = r.context)
        }
    }

    /**
     * Formats a partial Aadhaar string for in-progress display.
     *
     * Used by [validateProgressive]'s `Typing` state. Inserts spaces every 4 digits
     * up to the current length:
     * - `"1"` → `"1"`
     * - `"1234"` → `"1234"`
     * - `"12345"` → `"1234 5"`
     * - `"12345678"` → `"1234 5678"`
     * - `"123456789"` → `"1234 5678 9"`
     */
    private fun formatPartial(digits: String): String =
        digits.chunked(4).joinToString(" ")
```

public fun validate(value: String): ValidationResult {
    val normalized = normalize(value)
    if (normalized.isBlank()) return ValidationResult.Invalid(InvalidReason.EMPTY)
    if (normalized.length != EXPECTED_LENGTH) {
        return ValidationResult.Invalid(
            InvalidReason.WRONG_LENGTH,
            ValidationContext.LengthMismatch(expected = EXPECTED_LENGTH, actual = normalized.length),
        )
    }
    if (!DIGIT_REGEX.matches(normalized)) {
        // Find first non-digit position for the context.
        val pos = normalized.indexOfFirst { it !in '0'..'9' }
        return ValidationResult.Invalid(
            InvalidReason.INVALID_FORMAT,
            ValidationContext.CharacterAt(position = pos, char = normalized[pos]),
        )
    }
    if (normalized[0] in INVALID_PREFIX_DIGITS) {
        return ValidationResult.Invalid(
            InvalidReason.INVALID_PREFIX,
            ValidationContext.PrefixMismatch(expected = setOf('2', '3', '4', '5', '6', '7', '8', '9'), actual = normalized[0]),
        )
    }
    if (!Verhoeff.isValid(normalized)) {
        // Note: Aadhaar's checksum is a Verhoeff over all 12 digits — there isn't a discrete
        // "computed vs provided" character pair. Use ValidationContext.None for this case.
        return ValidationResult.Invalid(InvalidReason.INVALID_CHECKSUM)
    }
    return ValidationResult.Valid
}

```kotlin
// Copyright (c) The KotIndia Authors. Licensed under the Apache License, Version 2.0.
// SPDX-License-Identifier: Apache-2.0

package io.github.kotindia

private val DIN_REGEX: Regex = Regex("^\\d{8}$")
private const val DIN_EXPECTED_LENGTH: Int = 8

/**
 * Validator for Director Identification Number (DIN) issued by the Ministry of
 * Corporate Affairs (MCA), Government of India.
 *
 * DIN is required for any individual intending to be appointed as a director of a
 * company in India under the Companies Act, 2013.
 *
 * **Format:** 8 digits. No checksum (MCA does not publish a public algorithm).
 * Validation here is format-only — to verify a DIN is currently active, a network
 * call to MCA's DIN portal is required, which is out of scope for this library.
 *
 * **Note:** A DIN with all zeros (`"00000000"`) is technically format-valid but
 * is never issued in practice. We accept it for format consistency.
 *
 * This object is stateless and thread-safe.
 *
 * @sample io.github.kotindia.samples.dinSample
 */
public object DIN {
    /** The exact length a valid DIN must have. */
    public val maxLength: Int get() = DIN_EXPECTED_LENGTH

    /** The set of characters allowed in a DIN. DIN is digits-only. */
    public val allowedChars: Set<Char> get() = ('0'..'9').toSet()

    public fun validate(value: String): ValidationResult {
        val normalized = value.trim()
        if (normalized.isBlank()) return ValidationResult.Invalid(InvalidReason.EMPTY)
        if (normalized.length != DIN_EXPECTED_LENGTH) {
            return ValidationResult.Invalid(
                InvalidReason.WRONG_LENGTH,
                ValidationContext.LengthMismatch(expected = DIN_EXPECTED_LENGTH, actual = normalized.length),
            )
        }
        if (!DIN_REGEX.matches(normalized)) {
            val pos = normalized.indexOfFirst { it !in allowedChars }
            return ValidationResult.Invalid(
                InvalidReason.INVALID_FORMAT,
                ValidationContext.CharacterAt(position = pos, char = normalized[pos]),
            )
        }
        return ValidationResult.Valid
    }

    public fun isValid(value: String): Boolean = validate(value) is ValidationResult.Valid

    public fun sanitize(rawInput: String): String =
        rawInput.filter { it in allowedChars }.take(maxLength)

    public fun validateProgressive(value: String): ProgressiveResult {
        val normalized = value.trim()
        if (normalized.isEmpty()) return ProgressiveResult.Empty
        if (normalized.any { it !in allowedChars }) {
            val pos = normalized.indexOfFirst { it !in allowedChars }
            return ProgressiveResult.Invalid(
                reason = InvalidReason.INVALID_FORMAT,
                context = ValidationContext.CharacterAt(position = pos, char = normalized[pos]),
            )
        }
        if (normalized.length > maxLength) {
            return ProgressiveResult.Invalid(
                reason = InvalidReason.WRONG_LENGTH,
                context = ValidationContext.LengthMismatch(expected = maxLength, actual = normalized.length),
            )
        }
        if (normalized.length < maxLength) {
            return ProgressiveResult.Typing(visualText = normalized)
        }
        return when (val r = validate(normalized)) {
            is ValidationResult.Valid -> ProgressiveResult.Valid
            is ValidationResult.Invalid -> ProgressiveResult.Invalid(reason = r.reason, context = r.context)
        }
    }

    public fun validateAndFormat(value: String): ValidatedFormat {
        val result = validate(value)
        return when (result) {
            is ValidationResult.Valid -> ValidatedFormat(
                raw = value,
                formatted = value.trim(), // DIN has no canonical formatted form distinct from raw
                masked = null, // DIN is not PII (it's a public director ID, listed on MCA portal)
                result = result,
            )
            is ValidationResult.Invalid -> ValidatedFormat(
                raw = value,
                formatted = null,
                masked = null,
                result = result,
            )
        }
    }
}
```
